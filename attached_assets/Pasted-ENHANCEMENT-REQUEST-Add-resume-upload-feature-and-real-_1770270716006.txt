import OpenAI from 'openai'
import mammoth from 'mammoth'
import pdf from 'pdf-parse'

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
})

export async function extractTextFromPDF(buffer) {
  try {
    const data = await pdf(buffer)
    return data.text
  } catch (error) {
    console.error('PDF parsing error:', error)
    throw new Error('Failed to parse PDF')
  }
}

export async function extractTextFromDOCX(buffer) {
  try {
    const result = await mammoth.extractRawText({ buffer })
    return result.value
  } catch (error) {
    console.error('DOCX parsing error:', error)
    throw new Error('Failed to parse DOCX')
  }
}

export async function parseResumeWithAI(resumeText) {
  const completion = await openai.chat.completions.create({
    model: "gpt-4",
    messages: [{
      role: "system",
      content: `You are a resume parser for legal AI job matching. Extract structured information from the resume.

Return JSON with:
{
  "name": "Full name",
  "email": "Email if found",
  "phone": "Phone if found",
  "summary": "Brief professional summary (2-3 sentences)",
  "experience": [
    {
      "title": "Job title",
      "company": "Company name",
      "duration": "2020-2023",
      "description": "What they did"
    }
  ],
  "totalYearsExperience": 5,
  "skills": ["skill1", "skill2"],
  "education": [
    {
      "degree": "Degree name",
      "institution": "School",
      "year": "2020"
    }
  ],
  "preferredRoles": ["Product Manager", "Legal Engineer"],
  "preferredLocations": ["San Francisco", "Remote"],
  "desiredSalary": { "min": 120000, "max": 180000 },
  "isOpenToRemote": true,
  "legalBackground": true,
  "techBackground": true
}`
    }, {
      role: "user",
      content: `Parse this resume:\n\n${resumeText}`
    }],
    response_format: { type: "json_object" }
  })

  return JSON.parse(completion.choices[0].message.content)
}

export async function generateSearchQueryFromResume(parsedData) {
  const completion = await openai.chat.completions.create({
    model: "gpt-4",
    messages: [{
      role: "system",
      content: "Convert resume data into a natural language job search query."
    }, {
      role: "user",
      content: `Based on this resume data: ${JSON.stringify(parsedData)}
      
Generate a natural search query (2-3 sentences) that captures what this person is looking for.`
    }]
  })

  return completion.choices[0].message.content
}

import { NextResponse } from 'next/server'
import { getServerSession } from 'next-auth'
import { query } from '@/lib/db'
import { 
  extractTextFromPDF, 
  extractTextFromDOCX, 
  parseResumeWithAI,
  generateSearchQueryFromResume 
} from '@/lib/resume-parser'

export async function POST(request) {
  try {
    const session = await getServerSession()
    if (!session?.user?.email) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
    }

    const formData = await request.formData()
    const file = formData.get('resume')

    if (!file) {
      return NextResponse.json({ error: 'No file provided' }, { status: 400 })
    }

    // Check file type
    const fileType = file.type
    const validTypes = [
      'application/pdf',
      'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
      'application/msword'
    ]

    if (!validTypes.includes(fileType)) {
      return NextResponse.json(
        { error: 'Invalid file type. Please upload PDF or DOCX.' },
        { status: 400 }
      )
    }

    // Check file size (max 5MB)
    if (file.size > 5 * 1024 * 1024) {
      return NextResponse.json(
        { error: 'File too large. Max size is 5MB.' },
        { status: 400 }
      )
    }

    // Convert to buffer
    const bytes = await file.arrayBuffer()
    const buffer = Buffer.from(bytes)

    // Extract text based on file type
    let resumeText
    if (fileType === 'application/pdf') {
      resumeText = await extractTextFromPDF(buffer)
    } else {
      resumeText = await extractTextFromDOCX(buffer)
    }

    // Parse resume with AI
    const parsedData = await parseResumeWithAI(resumeText)

    // Generate search query from resume
    const searchQuery = await generateSearchQueryFromResume(parsedData)

    // Save to database
    await query(
      `UPDATE users 
       SET resume_text = $1, resume_filename = $2, extracted_data = $3
       WHERE email = $4`,
      [resumeText, file.name, JSON.stringify(parsedData), session.user.email]
    )

    return NextResponse.json({
      success: true,
      parsedData,
      searchQuery,
      message: 'Resume uploaded and parsed successfully'
    })

  } catch (error) {
    console.error('Resume upload error:', error)
    return NextResponse.json(
      { error: 'Failed to process resume' },
      { status: 500 }
    )
  }
}

// Get user's resume data
export async function GET(request) {
  try {
    const session = await getServerSession()
    if (!session?.user?.email) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
    }

    const result = await query(
      'SELECT resume_filename, extracted_data FROM users WHERE email = $1',
      [session.user.email]
    )

    if (result.rows.length === 0) {
      return NextResponse.json({ hasResume: false })
    }

    return NextResponse.json({
      hasResume: !!result.rows[0].resume_filename,
      filename: result.rows[0].resume_filename,
      extractedData: result.rows[0].extracted_data
    })

  } catch (error) {
    console.error('Error fetching resume:', error)
    return NextResponse.json(
      { error: 'Failed to fetch resume data' },
      { status: 500 }
    )
  }
}

  const [uploadProgress, setUploadProgress] = useState(0)

  const [resumeData, setResumeData] = useState(null)
  const fileInputRef = useRef(null)

  // Check if user has uploaded resume

    checkExistingResume()
  }, [])

  const checkExistingResume = async () => {
    try {
      const response = await fetch('/api/resume/upload')
      const data = await response.json()
      if (data.hasResume) {
        setUploadedResume(data.filename)
        setResumeData(data.extractedData)
      }
    } catch (error) {
      console.error('Error checking resume:', error)
    }

    e.preventDefault()

      onSearch(query)

    const file = e.target.files?.[0]
    if (!file) return

    setUploadProgress(0)

    const progressInterval = setInterval(() => {
      setUploadProgress(prev => {
        if (prev >= 90) {
          clearInterval(progressInterval)
          return 90
        }
        return prev + 10
      })

      const formData = new FormData()

        body: formData

        setUploadProgress(100)
        setUploadedResume(file.name)

          onSearch(data.searchQuery)

'use client'

import { useState } from 'react'

export default function JobCard({ job, matchScore = null, resumeData = null }) {
  const [isExpanded, setIsExpanded] = useState(false)

  const formatSalary = (min, max) => {
    if (!min && !max) return null
    const formatNum = (num) => `$${(num / 1000).toFixed(0)}K`
    if (min && max) return `${formatNum(min)} - ${formatNum(max)}`
    if (min) return `${formatNum(min)}+`
    return `Up to ${formatNum(max)}`
  }

  const getTimeAgo = (date) => {
    const days = Math.floor((new Date() - new Date(date)) / (1000 * 60 * 60 * 24))
    if (days === 0) return 'Today'
    if (days === 1) return '1d ago'
    if (days < 7) return `${days}d ago`
    if (days < 30) return `${Math.floor(days / 7)}w ago`
    return `${Math.floor(days / 30)}m ago`
  }

  const getMatchReasons = () => {
    if (!resumeData || !matchScore) return []
    
    const reasons = []
    
    if (resumeData.preferredRoles?.some(role => 
      job.title.toLowerCase().includes(role.toLowerCase())
    )) {
      reasons.push('Role matches your experience')
    }
    
    if (resumeData.isOpenToRemote && job.is_remote) {
      reasons.push('Remote position')
    }
    
    if (job.experience_min && job.experience_max && resumeData.totalYearsExperience) {
      if (resumeData.totalYearsExperience >= job.experience_min && 
          resumeData.totalYearsExperience <= job.experience_max) {
        reasons.push('Experience level matches')
      }
    }
    
    const matchingSkills = resumeData.skills?.filter(skill =>
      job.description?.toLowerCase().includes(skill.toLowerCase()) ||
      job.requirements?.toLowerCase().includes(skill.toLowerCase())
    )
    
    if (matchingSkills?.length > 0) {
      reasons.push(`${matchingSkills.length} matching skills`)
    }
    
    return reasons
  }

  const matchReasons = getMatchReasons()
  const salary = formatSalary(job.salary_min, job.salary_max)

  return (
    <div className="bg-white border border-border rounded-xl p-6 hover:border-primary hover:shadow-md transition-all duration-200 cursor-pointer group">
      <div className="flex flex-col md:flex-row gap-6">
        {/* Left Section */}
        <div className="flex-1">
          {/* Top Row */}
          <div className="flex items-center gap-3 mb-2">
            {job.company_logo && (
              <img 
                src={job.company_logo} 
                alt={job.company}
                className="w-10 h-10 rounded-lg"
                onError={(e) => {
                  e.target.style.display = 'none'
                }}
              />
            )}
            <span className="text-sm font-medium text-text-secondary">
              {job.company}
            </span>
            <span className="text-xs text-text-tertiary">
              • {getTimeAgo(job.posted_date)}
            </span>
          </div>

          {/* Job Title */}
          <h3 className="text-xl font-medium text-text-primary mb-3 group-hover:text-primary transition-colors">
            {job.title}
          </h3>

          {/* Tags */}
          <div className="flex flex-wrap gap-2 mb-3">
            {job.role_type && (
              <span className="px-3 py-1.5 bg-orange-50 text-primary rounded-lg text-xs font-medium">
                {job.role_type}
              </span>
            )}
            
            {job.is_remote ? (
              <span className="px-3 py-1.5 bg-green-50 text-green-700 rounded-lg text-xs font-medium">
                🌍 Remote
              </span>
            ) : job.location && (
              <span className="px-3 py-1.5 bg-blue-50 text-blue-700 rounded-lg text-xs font-medium">
                📍 {job.location}
              </span>
            )}
            
            {job.experience_min && job.experience_max && (
              <span className="px-3 py-1.5 bg-surface text-text-secondary rounded-lg text-xs font-medium">
                {job.experience_min}-{job.experience_max} years
              </span>
            )}

            {salary && (
              <span className="px-3 py-1.5 bg-green-50 text-green-700 rounded-lg text-xs font-medium">
                💰 {salary}
              </span>
            )}
          </div>

          {/* Description - Expandable */}

UPDATED FILE: app/page.js (WITH RESUME SUPPORT)

═══════════════════════════════════════════════════════════════════════════

'use client'

import { useState, useEffect } from 'react'
import { useSession } from 'next-auth/react'
import Header from '@/components/Header'
import SearchBar from '@/components/SearchBar'
import JobList from '@/components/JobList'

export default function HomePage() {
  const { data: session, status } = useSession()
  const [jobs, setJobs] = useState([])
  const [matchScores, setMatchScores] = useState(null)
  const [isLoading, setIsLoading] = useState(true)
  const [isSearching, setIsSearching] = useState(false)
  const [hasSearched, setHasSearched] = useState(false)
  const [resumeData, setResumeData] = useState(null)
  const [resultCount, setResultCount] = useState(0)

  // Load all jobs on mount
  useEffect(() => {
    fetchAllJobs()
    checkResumeData()
  }, [])

  const checkResumeData = async () => {
    try {
      const response = await fetch('/api/resume/upload')
      const data = await response.json()
      if (data.hasResume) {
        setResumeData(data.extractedData)
      }
    } catch (error) {
      console.error('Error fetching resume:', error)
    }
  }

  const fetchAllJobs = async () => {
    setIsLoading(true)
    try {
      const response = await fetch('/api/jobs')
      const data = await response.json()
      setJobs(data)
      setResultCount(data.length)
      setMatchScores(null)
      setHasSearched(false)
    } catch (error) {
      console.error('Error fetching jobs:', error)
    } finally {
      setIsLoading(false)
    }
  }

  const handleSearch = async (query) => {
    setIsSearching(true)
    setHasSearched(true)
    
    try {
      const response = await fetch('/api/search', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ query })
      })

      const data = await response.json()
      
      if (data.jobs) {
        setJobs(data.jobs)
        setResultCount(data.jobs.length)
        setMatchScores(data.jobs.map(j => j.matchScore))
      }
    } catch (error) {
      console.error('Search error:', error)
    } finally {
      setIsSearching(false)
    }
  }

  if (status === 'loading') {
    return (
      <div className="min-h-screen flex items-center justify-center bg-surface">
        <div className="text-center">
          <div className="inline-block animate-spin rounded-full h-12 w-12 border-4 border-primary border-t-transparent"></div>
          <p className="mt-4 text-text-secondary">Loading...</p>
        </div>
      </div>
    )
  }

  return (
    <div className="min-h-screen bg-white">
      <Header />

      {/* Hero Section */}
      <section className="bg-gradient-to-b from-white via-orange-50/20 to-surface py-16 px-6">
        <div className="max-w-[800px] mx-auto text-center">
          <h1 className="text-5xl font-medium text-text-primary mb-3 leading-tight">
            Find your next role in legal AI
          </h1>
          <p className="text-lg text-text-secondary mb-8">
            Upload your resume or describe what you're looking for
          </p>

          <SearchBar 
            onSearch={handleSearch} 
            isLoading={isSearching}
          />

          <div className="mt-6 flex items-center justify-center gap-6 text-xs text-text-tertiary">
            <span className="flex items-center gap-1">
              <span className="w-2 h-2 bg-green-500 rounded-full animate-pulse"></span>
              {resultCount} active jobs
            </span>
            <span>•</span>
            <span>127 companies</span>
            <span>•</span>
            <span>Updated daily</span>
          </div>
        </div>
      </section>

      {/* Results Section */}
      <section className="bg-surface py-12 px-6 min-h-screen">
        <div className="max-w-[1120px] mx-auto">
          {/* Results Header */}
          <div className="flex items-center justify-between mb-6">
            <div>
              <h2 className="text-xl font-medium text-text-primary">
                {hasSearched ? (
                  <span>
                    {resultCount} {resultCount === 1 ? 'job matches' : 'jobs match'} your search
                  </span>
                ) : (
                  'All jobs'
                )}
              </h2>
              {hasSearched && resultCount > 0 && (
                <p className="text-sm text-text-secondary mt-1">
                  Sorted by relevance
                </p>
              )}
            </div>
            
            {hasSearched && (
              <button
                onClick={fetchAllJobs}
                className="text-sm text-primary hover:underline flex items-center gap-1"
              >
                ✕ Clear search
              </button>
            )}
          </div>

          {/* Job List */}
          <JobList 
            jobs={jobs} 
            matchScores={matchScores}
            resumeData={resumeData}
            isLoading={isLoading || isSearching} 
          />
        </div>
      </section>

      {/* Footer */}
      <footer className="bg-white border-t border-border py-8 px-6">
        <div className="max-w-[1120px] mx-auto flex flex-col md:flex-row justify-between items-center gap-4">
          <div className="text-center md:text-left">
            <p className="text-sm font-medium text-text-primary">Legal AI Careers</p>
            <p className="text-xs text-text-tertiary mt-1">
              Curated opportunities in legal technology
            </p>
          </div>
          
          <div className="flex gap-6 text-xs text-text-secondary">
            <a href="#" className="hover:text-primary transition-colors">About</a>
            <span>•</span>
            <a href="#" className="hover:text-primary transition-colors">Contact</a>
            <span>•</span>
            <a href="#" className="hover:text-primary transition-colors">Privacy</a>
            <span>•</span>
            <a href="#" className="hover:text-primary transition-colors">Terms</a>
          </div>
        </div>
      </footer>
    </div>
  )
}
