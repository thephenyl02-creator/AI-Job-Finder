CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  email VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  name VARCHAR(255),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Jobs table
CREATE TABLE jobs (
  id SERIAL PRIMARY KEY,
  title VARCHAR(255) NOT NULL,
  company VARCHAR(255) NOT NULL,

  is_remote BOOLEAN DEFAULT false,

  posted_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  is_active BOOLEAN DEFAULT true,
  embedding VECTOR(1536)  -- For semantic search (OpenAI embeddings)
);

CREATE INDEX ON jobs USING ivfflat (embedding vector_cosine_ops);

import OpenAI from 'openai'

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
})

export async function generateEmbedding(text) {
  const response = await openai.embeddings.create({
    model: "text-embedding-3-small",
    input: text,
  })
  return response.data[0].embedding
}

export async function semanticSearch(searchQuery) {
  // Generate embedding for search query
  const queryEmbedding = await generateEmbedding(searchQuery)
  
  // Use GPT-4 to extract search parameters
  const completion = await openai.chat.completions.create({
    model: "gpt-4",
    messages: [{
      role: "system",
      content: `You are a job search assistant. Extract structured information from user queries.
      Return JSON with: roleType, experienceMin, experienceMax, location, isRemote, salaryMin, salaryMax.
      If not mentioned, set to null.`
    }, {
      role: "user",
      content: searchQuery
    }],
    response_format: { type: "json_object" }
  })
  
  const searchParams = JSON.parse(completion.choices[0].message.content)
  
  return {
    embedding: queryEmbedding,
    params: searchParams
  }
}

export default openai

import { hash, compare } from 'bcryptjs'
import { query } from './db'

export async function hashPassword(password) {
  return await hash(password, 12)
}

export async function verifyPassword(password, hashedPassword) {
  return await compare(password, hashedPassword)
}

export async function createUser(email, password) {
  const hashedPassword = await hashPassword(password)
  
  const result = await query(
    'INSERT INTO users (email, password_hash) VALUES ($1, $2) RETURNING id, email',
    [email, hashedPassword]
  )
  
  return result.rows[0]
}

export async function getUserByEmail(email) {
  const result = await query(
    'SELECT * FROM users WHERE email = $1',
    [email]
  )
  
  return result.rows[0]
}

import NextAuth from "next-auth"
import CredentialsProvider from "next-auth/providers/credentials"
import { getUserByEmail, verifyPassword, createUser } from "@/lib/auth"

const handler = NextAuth({
  providers: [
    CredentialsProvider({
      name: 'Credentials',
      credentials: {
        email: { label: "Email", type: "email" },
        password: { label: "Password", type: "password" }
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) {
          throw new Error('Please enter email and password')
        }

        let user = await getUserByEmail(credentials.email)
        
        // For demo: Auto-create user if doesn't exist
        if (!user) {
          user = await createUser(credentials.email, credentials.password)
        }

        const isValid = await verifyPassword(
          credentials.password,
          user.password_hash
        )

        if (!isValid) {
          throw new Error('Invalid password')
        }

        return {
          id: user.id,
          email: user.email,
        }
      }
    })
  ],
  session: {
    strategy: "jwt",
  },
  pages: {
    signIn: '/login',
  },
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id
      }
      return token
    },
    async session({ session, token }) {
      if (token) {
        session.user.id = token.id
      }
      return session
    },
  },
})

export { handler as GET, handler as POST }

export default function JobCard({ job, matchScore = null }) {
  const formatSalary = (min, max) => {
    if (!min && !max) return 'Salary not disclosed'
    const formatNum = (num) => `$${(num / 1000).toFixed(0)}K`
    if (min && max) return `${formatNum(min)} - ${formatNum(max)}`
    if (min) return `${formatNum(min)}+`
    return `Up to ${formatNum(max)}`
  }

  const getTimeAgo = (date) => {
    const days = Math.floor((new Date() - new Date(date)) / (1000 * 60 * 60 * 24))
    if (days === 0) return 'Today'
    if (days === 1) return '1d ago'
    if (days < 7) return `${days}d ago`
    if (days < 30) return `${Math.floor(days / 7)}w ago`
    return `${Math.floor(days / 30)}m ago`
  }

  return (
    <div className="card hover:border-primary cursor-pointer group">
      <div className="flex flex-col md:flex-row gap-6">
        {/* Left Section */}
        <div className="flex-1">
          {/* Company Info */}
          <div className="flex items-center gap-3 mb-2">
            {job.company_logo && (
              <img 
                src={job.company_logo} 
                alt={job.company}
                className="w-10 h-10 rounded-lg"
              />
            )}
            <span className="text-sm font-medium text-text-secondary">
              {job.company}
            </span>
            <span className="text-xs text-text-tertiary">
              {getTimeAgo(job.posted_date)}
            </span>
          </div>

          {/* Job Title */}
          <h3 className="text-xl font-medium text-text-primary mb-3 group-hover:text-primary transition-colors">
            {job.title}
          </h3>

import { query } from '@/lib/db'
import { semanticSearch } from '@/lib/openai'
import { NextResponse } from 'next/server'

export async function POST(request) {
  try {
    const { query: searchQuery } = await request.json()

    if (!searchQuery || searchQuery.trim().length === 0) {
      return NextResponse.json(
        { error: 'Search query is required' },
        { status: 400 }
      )
    }

    // Get semantic search results
    const { embedding, params } = await semanticSearch(searchQuery)

    // Build SQL query with filters from AI extraction
    let sql = 'SELECT * FROM jobs WHERE is_active = true'
    const queryParams = []
    let paramCount = 1

    if (params.roleType) {
      sql += ` AND LOWER(role_type) LIKE LOWER($${paramCount})`
      queryParams.push(`%${params.roleType}%`)
      paramCount++
    }

    if (params.isRemote !== null) {
      sql += ` AND is_remote = $${paramCount}`
      queryParams.push(params.isRemote)
      paramCount++
    }

    if (params.experienceMin !== null) {
      sql += ` AND experience_max >= $${paramCount}`
      queryParams.push(params.experienceMin)
      paramCount++
    }

    if (params.experienceMax !== null) {
      sql += ` AND experience_min <= $${paramCount}`
      queryParams.push(params.experienceMax)
      paramCount++
    }

    if (params.salaryMin !== null) {
      sql += ` AND salary_max >= $${paramCount}`
      queryParams.push(params.salaryMin)
      paramCount++
    }

    sql += ' ORDER BY posted_date DESC LIMIT 50'

    const result = await query(sql, queryParams)

    // Calculate simple match scores (0-100)
    // In production, you'd use vector similarity with embeddings
    const jobsWithScores = result.rows.map(job => {
      let score = 70 // Base score
      
      if (params.roleType && job.role_type?.toLowerCase().includes(params.roleType.toLowerCase())) {
        score += 15
      }
      
      if (params.isRemote === job.is_remote) {
        score += 10
      }
      
      if (params.experienceMin && params.experienceMax) {
        const jobExpMid = (job.experience_min + job.experience_max) / 2
        const userExpMid = (params.experienceMin + params.experienceMax) / 2
        if (Math.abs(jobExpMid - userExpMid) <= 2) {
          score += 5
        }
      }

      return {
        ...job,
        matchScore: Math.min(score, 98) // Cap at 98%
      }
    })

    // Sort by match score
    jobsWithScores.sort((a, b) => b.matchScore - a.matchScore)

    return NextResponse.json({
      jobs: jobsWithScores,
      searchParams: params
    })
  } catch (error) {
    console.error('Search error:', error)
    return NextResponse.json(
      { error: 'Search failed' },
      { status: 500 }
    )
  }
}
